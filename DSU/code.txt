class DSU {
	int p[nax], sz[nax];
public :
	DSU() {
		for(int i = 0; i < nax; ++i) {
			p[i] = i;
			sz[i] = 1;
		}
	}
	int get(int x) {
		return x == p[x] ? x : x = get(p[x]);
	}
	int unite(int x, int y) {
		x = get(x);
		y = get(y);
		if(x == y) {
			return 1;
		}
		if(sz[x] > sz[y]) {
			swap(x, y);
		}
		p[x] = y;
		sz[y] += sz[x];
		return 0;
	}
};
